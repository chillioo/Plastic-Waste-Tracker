import java.io.*;
import java.util.*;

public class PlasticWasteTracker {

    static class PlasticWasteRecord {
        private String country;
        private int year;
        private double plasticWaste;

        public PlasticWasteRecord(String country, int year, double plasticWaste)
        {
            this.country = country;
            this.year = year;
            this.plasticWaste = plasticWaste;
        }

        public String getCountry() { return country; }
        public int getYear() { return year; }
        public double getPlasticWaste() { return plasticWaste; }

        public String toString() {
            return country + " | " + year + " | " + plasticWaste;
        }
    }

    static class PlasticDataProcessor {
        private List<PlasticWasteRecord> records = new ArrayList<>();

        public void loadData(String filePath) throws IOException
        {
            records.clear();
            try (BufferedReader br = new BufferedReader(new FileReader(filePath))) {
                String line = br.readLine(); // skip header
                if (line == null) throw new IOException("Empty file or missing header");

                while ((line = br.readLine()) != null) {
                    String[] parts = line.split(",");
                    if (parts.length != 3) continue; // skip invalid lines
                    String country = parts[0].trim();
                    int year = Integer.parseInt(parts[1].trim());
                    double plasticWaste = Double.parseDouble(parts[2].trim());
                    records.add(new PlasticWasteRecord(country, year, plasticWaste));
                }
            }
        }

        public List<PlasticWasteRecord> getDataByCountry(String country) {
            List<PlasticWasteRecord> result = new ArrayList<>();
            for (PlasticWasteRecord r : records) {
                if (r.getCountry().equalsIgnoreCase(country)) {
                    result.add(r);
                }
            }
            return result;
        }

        public List<PlasticWasteRecord> getAllRecords() {
            return records;
        }
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        PlasticDataProcessor processor = new PlasticDataProcessor();

        System.out.println("=== Plastic Waste Data Tracker ===");
        while (true) {
            System.out.println("\nMenu:");
            System.out.println("1. Upload CSV data file");
            System.out.println("2. Search records by country");
            System.out.println("3. Show all records");
            System.out.println("4. Exit");
            System.out.print("Choose an option: ");

            String choice = scanner.nextLine();

            switch (choice) {
                case "1":
                    System.out.print("Enter full path to your CSV file: ");
                    String path = scanner.nextLine();
                    try {
                        processor.loadData(path);
                        System.out.println("Data loaded successfully! Records count: " + processor.getAllRecords().size());
                    } catch (IOException e) {
                        System.out.println("Error loading data: " + e.getMessage());
                    } catch (NumberFormatException e) {
                        System.out.println("Invalid number format in CSV.");
                    }
                    break;

                case "2":
                    System.out.print("Enter country to search: ");
                    String country = scanner.nextLine();
                    List<PlasticWasteRecord> found = processor.getDataByCountry(country);
                    if (found.isEmpty()) {
                        System.out.println("No records found for " + country);
                    } else {
                        System.out.println("Records for " + country + ":");
                        for (PlasticWasteRecord r : found) {
                            System.out.println(r);
                        }
                    }
                    break;

                case "3":
                    List<PlasticWasteRecord> all = processor.getAllRecords();
                    if (all.isEmpty()) {
                        System.out.println("No records loaded yet.");
                    } else {
                        System.out.println("All loaded records:");
                        for (PlasticWasteRecord r : all) {
                            System.out.println(r);
                        }
                    }
                    break;

                case "4":
                    System.out.println("Exiting program.");
                    scanner.close();
                    return;

                default:
                    System.out.println("Invalid option, try again.");
            }
        }
    }
}
